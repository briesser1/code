CREATE TABLE jurisdiction_economics (
    economics_id INTEGER PRIMARY KEY,
    jurisdiction_id INTEGER NOT NULL UNIQUE,

    -- Population data
    population INTEGER,
    population_year INTEGER DEFAULT 2024,
    population_source TEXT,

    -- Economic indicators
    gdp_per_capita INTEGER,  -- in USD
    gdp_per_capita_year INTEGER DEFAULT 2024,
    gdp_total_millions INTEGER,  -- total GDP in millions USD

    -- Government budget data
    total_budget_millions INTEGER,  -- total state/territory/tribal budget in millions USD
    budget_fiscal_year INTEGER,
    general_fund_millions INTEGER,  -- general fund portion

    -- Social services spending
    social_services_spending_millions INTEGER,  -- public welfare/human services spending
    social_services_pct_of_budget REAL,  -- percentage of total budget
    public_welfare_per_capita INTEGER,  -- per capita welfare spending

    -- Data quality
    data_quality TEXT CHECK(data_quality IN ('Complete', 'Partial', 'Estimated', 'Limited')),
    notes TEXT,
    last_updated DATE DEFAULT CURRENT_DATE,

    FOREIGN KEY (jurisdiction_id) REFERENCES jurisdictions(jurisdiction_id)
);

CREATE VIEW v_state_systems AS
SELECT
    j.name,
    j.abbreviation,
    jt.type_name as jurisdiction_type,
    j.current_system_name,
    st.type_code as system_type,
    j.ccwis_status,
    j.ccwis_declaration_date,
    j.administering_agency,
    j.county_administered
FROM jurisdictions j
JOIN jurisdiction_types jt ON j.jurisdiction_type_id = jt.type_id
LEFT JOIN system_types st ON j.current_system_type_id = st.type_id
WHERE j.jurisdiction_type_id = 1
ORDER BY j.name;
CREATE VIEW v_ccwis_summary AS
SELECT
    ccwis_status,
    COUNT(*) as count,
    GROUP_CONCAT(abbreviation, ', ') as jurisdictions
FROM jurisdictions
WHERE jurisdiction_type_id IN (1, 2, 3)
GROUP BY ccwis_status;
CREATE VIEW v_planned_implementations AS
SELECT
    j.name as jurisdiction,
    j.current_system_name,
    p.planned_system_name,
    p.target_ccwis_status,
    p.project_status,
    p.planned_go_live_date,
    p.estimated_cost
FROM planned_systems p
JOIN jurisdictions j ON p.jurisdiction_id = j.jurisdiction_id
WHERE p.project_status NOT IN ('Operational', 'Cancelled')
ORDER BY p.planned_go_live_date;
CREATE VIEW v_platform_usage AS
SELECT
    p.platform_name,
    p.platform_category,
    v.vendor_name,
    COUNT(DISTINCT jp.jurisdiction_id) as jurisdiction_count,
    GROUP_CONCAT(j.abbreviation, ', ') as jurisdictions
FROM technology_platforms p
LEFT JOIN jurisdiction_platforms jp ON p.platform_id = jp.platform_id
LEFT JOIN jurisdictions j ON jp.jurisdiction_id = j.jurisdiction_id
LEFT JOIN vendors v ON p.vendor_id = v.vendor_id
GROUP BY p.platform_id
HAVING jurisdiction_count > 0
ORDER BY jurisdiction_count DESC;
CREATE VIEW v_jurisdiction_tech_stack AS
SELECT
    j.name as jurisdiction,
    j.abbreviation,
    j.current_system_name,
    j.ccwis_status,
    GROUP_CONCAT(p.platform_name || ' (' || jp.role || ')', '; ') as tech_stack,
    GROUP_CONCAT(DISTINCT v.vendor_name) as vendors
FROM jurisdictions j
LEFT JOIN jurisdiction_platforms jp ON j.jurisdiction_id = jp.jurisdiction_id
LEFT JOIN technology_platforms p ON jp.platform_id = p.platform_id
LEFT JOIN vendors v ON jp.vendor_id = v.vendor_id OR p.vendor_id = v.vendor_id
WHERE j.jurisdiction_type_id = 1
GROUP BY j.jurisdiction_id
ORDER BY j.name;
CREATE VIEW v_cloud_adoption AS
SELECT
    CASE
        WHEN EXISTS (SELECT 1 FROM jurisdiction_platforms jp2
                     JOIN technology_platforms p2 ON jp2.platform_id = p2.platform_id
                     WHERE jp2.jurisdiction_id = j.jurisdiction_id
                     AND p2.platform_category = 'Cloud Infrastructure') THEN 'Cloud'
        ELSE 'On-Premises/Unknown'
    END as deployment_model,
    COUNT(*) as state_count
FROM jurisdictions j
WHERE j.jurisdiction_type_id = 1
GROUP BY deployment_model;
CREATE VIEW v_acf_ccwis_status AS

    p.project_status as replacement_status
FROM jurisdictions j
LEFT JOIN planned_systems p ON j.jurisdiction_id = p.jurisdiction_id
WHERE j.jurisdiction_type_id IN (1, 2, 3)
ORDER BY j.name;

CREATE VIEW v_economics_by_jurisdiction_type AS
SELECT
    jt.type_name as jurisdiction_type,
    COUNT(je.jurisdiction_id) as count,
    ROUND(AVG(je.population), 0) as avg_population,
    ROUND(AVG(je.gdp_per_capita), 0) as avg_gdp_per_capita,
    ROUND(AVG(je.total_budget_millions), 0) as avg_budget_millions,
    ROUND(SUM(je.total_budget_millions), 0) as total_budget_millions,
    ROUND(AVG(je.social_services_spending_millions), 0) as avg_social_services_millions,
    ROUND(AVG(je.public_welfare_per_capita), 0) as avg_welfare_per_capita
FROM jurisdiction_economics je
JOIN jurisdictions j ON je.jurisdiction_id = j.jurisdiction_id
JOIN jurisdiction_types jt ON j.jurisdiction_type_id = jt.type_id
GROUP BY jt.type_id
ORDER BY total_budget_millions DESC;

        ELSE 'Very Small (<$10B)'
    END as budget_tier,
    COUNT(DISTINCT j.jurisdiction_id) as state_count,
    GROUP_CONCAT(DISTINCT v.vendor_name) as vendors_used,
    GROUP_CONCAT(DISTINCT p.platform_name) as platforms_used,
    ROUND(AVG(je.gdp_per_capita), 0) as avg_gdp_per_capita,
    ROUND(AVG(je.public_welfare_per_capita), 0) as avg_welfare_per_capita
FROM jurisdictions j
JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
LEFT JOIN jurisdiction_platforms jp ON j.jurisdiction_id = jp.jurisdiction_id
LEFT JOIN vendors v ON jp.vendor_id = v.vendor_id
LEFT JOIN technology_platforms p ON jp.platform_id = p.platform_id
WHERE j.jurisdiction_type_id = 1  -- States only
GROUP BY budget_tier
ORDER BY
    CASE budget_tier
        WHEN 'Very Large (>$100B)' THEN 1
        WHEN 'Large ($50B-$100B)' THEN 2
        WHEN 'Medium ($25B-$50B)' THEN 3
        WHEN 'Small ($10B-$25B)' THEN 4
        ELSE 5
    END
;
CREATE VIEW v_vendor_by_jurisdiction_type AS
SELECT
    jt.type_name as jurisdiction_type,
    v.vendor_name,
    v.vendor_type,
    COUNT(DISTINCT jp.jurisdiction_id) as jurisdiction_count,
    GROUP_CONCAT(DISTINCT j.abbreviation) as jurisdictions,
    ROUND(AVG(je.total_budget_millions), 0) as avg_budget_millions,
    ROUND(AVG(je.gdp_per_capita), 0) as avg_gdp_per_capita,
    ROUND(AVG(je.population), 0) as avg_population
FROM jurisdiction_platforms jp
JOIN jurisdictions j ON jp.jurisdiction_id = j.jurisdiction_id
JOIN jurisdiction_types jt ON j.jurisdiction_type_id = jt.type_id
JOIN vendors v ON jp.vendor_id = v.vendor_id
LEFT JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
GROUP BY jt.type_id, v.vendor_id
ORDER BY jt.type_id, jurisdiction_count DESC;
CREATE VIEW v_platform_by_jurisdiction_type AS
SELECT
    jt.type_name as jurisdiction_type,
    p.platform_name,
    p.platform_category,
    COUNT(DISTINCT jp.jurisdiction_id) as adoption_count,
    GROUP_CONCAT(DISTINCT j.abbreviation) as jurisdictions,
    ROUND(AVG(je.total_budget_millions), 0) as avg_budget_millions,
    ROUND(AVG(je.gdp_per_capita), 0) as avg_gdp_per_capita
FROM jurisdiction_platforms jp
JOIN jurisdictions j ON jp.jurisdiction_id = j.jurisdiction_id
JOIN jurisdiction_types jt ON j.jurisdiction_type_id = jt.type_id
JOIN technology_platforms p ON jp.platform_id = p.platform_id
LEFT JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
GROUP BY jt.type_id, p.platform_id
ORDER BY jt.type_id, adoption_count DESC;
CREATE VIEW v_ccwis_status_economics AS
SELECT
    j.ccwis_status,
    COUNT(*) as count,
    ROUND(AVG(je.population), 0) as avg_population,
    ROUND(AVG(je.gdp_per_capita), 0) as avg_gdp_per_capita,
    ROUND(AVG(je.total_budget_millions), 0) as avg_budget_millions,
    ROUND(AVG(je.public_welfare_per_capita), 0) as avg_welfare_per_capita,
    GROUP_CONCAT(j.abbreviation, ', ') as jurisdictions
FROM jurisdictions j
JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
WHERE j.jurisdiction_type_id = 1  -- States only
GROUP BY j.ccwis_status
ORDER BY count DESC
;
CREATE VIEW v_jurisdiction_economics_summary AS
SELECT
    j.name,
    j.abbreviation,
    jt.type_name as jurisdiction_type,
    j.ccwis_status,
    j.current_system_name,
    je.population,
    je.gdp_per_capita,
    je.total_budget_millions,
    je.social_services_spending_millions,
    je.public_welfare_per_capita,
    je.data_quality,
    je.notes as economic_notes
FROM jurisdictions j
JOIN jurisdiction_types jt ON j.jurisdiction_type_id = jt.type_id
LEFT JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
ORDER BY jt.type_id, j.name;
CREATE VIEW v_states_by_budget_with_vendors AS
SELECT
    j.name as state,
    j.abbreviation,
    je.population,
    je.gdp_per_capita,
    je.total_budget_millions,
    je.social_services_spending_millions,
    je.public_welfare_per_capita,
    j.ccwis_status,
    j.current_system_name,
    GROUP_CONCAT(DISTINCT v.vendor_name) as vendors,
    GROUP_CONCAT(DISTINCT p.platform_name) as platforms
FROM jurisdictions j
JOIN jurisdiction_economics je ON j.jurisdiction_id = je.jurisdiction_id
LEFT JOIN jurisdiction_platforms jp ON j.jurisdiction_id = jp.jurisdiction_id
LEFT JOIN vendors v ON jp.vendor_id = v.vendor_id
LEFT JOIN technology_platforms p ON jp.platform_id = p.platform_id
WHERE j.jurisdiction_type_id = 1
GROUP BY j.jurisdiction_id
ORDER BY je.total_budget_millions DESC;
COMMIT;
